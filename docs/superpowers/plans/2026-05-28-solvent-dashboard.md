# Solvent Dashboard Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the Plan 4 MVP — a Next.js app at `web/` with a `/` landing page (Human-vs-AI scoreboard, narrative hero, how-it-works) and an `/app` dashboard (bento layout: hero stat + chart + policy + decision log + footer), rendered in the locked "Schematic Blueprint" visual identity from `docs/superpowers/specs/2026-05-28-solvent-dashboard-design.md` and pixel-aligned with `docs/superpowers/specs/2026-05-28-solvent-dashboard-mockup.html`. Mock data only — live-chain reads are deferred to integration phase.

**Architecture:** Single Next.js 15 App Router project under `web/`, parallel to `contracts/` and `agent/`. Pages: `web/src/app/page.tsx` (landing), `web/src/app/app/page.tsx` (dashboard). Visual primitives (`SchematicBackground`, `Panel`, `BrandMark`) under `web/src/components/`. Mock fixtures and the Plan-3 benchmark JSON reader under `web/src/lib/`. Palette/typography exposed as CSS custom properties in `globals.css`; layout via Tailwind utilities; component-specific styling inline where the mockup is inline (atmospheric gradients, decorative SVGs, IC corner notches). Static-exportable (`output: "export"` in `next.config.ts`).

**Tech Stack:** Next.js 15 (App Router), TypeScript strict, Tailwind CSS (utilities only), `next/font/google` for Inter + JetBrains Mono, inline SVG charts (no chart library), Vitest + jsdom + `@testing-library/react` for unit tests.

**Working branch:** `plan-4-dashboard` (created from `master`, merged back at end via `superpowers:finishing-a-development-branch`).

```typescript
import { readFile } from "node:fs/promises";
import { join } from "node:path";

export interface DecisionLogEntry {
  timestamp: number;
  regime: number;
  action: number;
  reasonCode: string;
  valueAfter: string; // bigint serialized as decimal string
}

export interface ScenarioResult {
  scenarioName: string;
  strategyName: string;
  initialValue: string;
  finalValue: string;
  pctPreservedBps: number;
  log: DecisionLogEntry[];
}

export interface BenchmarkScenario {
  name: string;
  description: string;
  results: ScenarioResult[];
}

export interface BenchmarkReport {
  generatedAt: number;
  scenarios: BenchmarkScenario[];
}

/** Load the snapshot. Node-only — used by tests and (via fetch) by client components. */
export async function loadBenchmark(): Promise<BenchmarkReport> {
  // In tests / server we read from disk; in the browser we'll fetch /benchmark-report.json.
  // For the unit test (vitest jsdom) the fs path is what matters.
  const path = join(process.cwd(), "public", "benchmark-report.json");
  const raw = await readFile(path, "utf8");
  return JSON.parse(raw) as BenchmarkReport;
}

/** Browser fetch variant — used inside React components. */
export async function fetchBenchmark(): Promise<BenchmarkReport> {
  const res = await fetch("/benchmark-report.json", { cache: "force-cache" });
  if (!res.ok) throw new Error(`benchmark fetch failed: ${res.status}`);
  return (await res.json()) as BenchmarkReport;
}

export interface HeadlineScores {
  ai: number; // percent
  human: number;
  hodl: number;
}

const STRATEGY_KEYS = {
  ai: "solvent-ai",
  human: "delayed-human",
  hodl: "passive-hodl",
} as const;

/** Pull a scenario's three strategy pctPreservedBps and convert to percent. */
export function headlineScores(report: BenchmarkReport, scenarioName: string): HeadlineScores {
  const scenario = report.scenarios.find((s) => s.name === scenarioName);
  if (!scenario) throw new Error(`scenario not found: ${scenarioName}`);
  const pick = (name: string): number => {
    const r = scenario.results.find((x) => x.strategyName === name);
    if (!r) throw new Error(`strategy not found in ${scenarioName}: ${name}`);
    return r.pctPreservedBps / 100;
  };
  return { ai: pick(STRATEGY_KEYS.ai), human: pick(STRATEGY_KEYS.human), hodl: pick(STRATEGY_KEYS.hodl) };
}
```

interface ScoreboardProps {
  ai: number;
  human: number;
  hodl: number;
}

interface Row {
  id: "ai" | "human" | "hodl";
  who: string;
  verb: string;
  value: string;
  emphasis: "high" | "mid" | "low";
}

function fmtPct(n: number): string {
  return Number.isInteger(n) ? `${n}%` : `${n.toFixed(1)}%`;
}

export default function Scoreboard({ ai, human, hodl }: ScoreboardProps) {
  const rows: Row[] = [
    { id: "ai", who: "AI", verb: "saved", value: fmtPct(ai), emphasis: "high" },
    { id: "human", who: "Human", verb: "kept", value: fmtPct(human), emphasis: "mid" },
    { id: "hodl", who: "HODL", verb: "rode to", value: fmtPct(hodl), emphasis: "low" },
  ];

  const colourFor = (e: Row["emphasis"]): string =>
    e === "high" ? "var(--ink-cyan)" : e === "mid" ? "var(--text-strong)" : "var(--warm-gold)";

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: 14 }}>
      {rows.map((r) => (
        <div
          key={r.id}
          data-row={r.id}
          className="mono"
          style={{
            display: "flex",
            alignItems: "baseline",
            gap: 14,
            fontSize: 18,
            color: "var(--text-body)",
            borderBottom: "1px solid var(--border-cyan-faint)",
            paddingBottom: 12,
          }}
        >
          <span style={{ minWidth: 80, color: "var(--text-muted)", fontSize: 12, letterSpacing: "0.16em", textTransform: "uppercase" }}>{r.who}</span>
          <span style={{ flex: 1, color: "var(--text-muted)", fontSize: 13 }}>· {r.verb} ·</span>
          <span
            data-value
            style={{
              color: colourFor(r.emphasis),
              fontSize: 28,
              fontWeight: 300,
              letterSpacing: "-0.01em",
            }}
          >
            {r.value}
          </span>
        </div>
      ))}
    </div>
  );
}

"use client";

import { useEffect, useState } from "react";
import Panel from "./Panel";
import { fetchBenchmark, type BenchmarkReport, type DecisionLogEntry } from "../lib/benchmark";

const REASON_COLOUR: Record<string, string> = {
  "park-calm": "var(--ink-cyan-bright)",

  "bridge-holding": "var(--text-strong)",
  "swap-safe": "var(--ink-cyan)",
  "remain-safe": "var(--text-strong)",
  "observe": "var(--warm-gold)",
};

export default function BenchmarkReplay() {
  const [report, setReport] = useState<BenchmarkReport | null>(null);

  useEffect(() => {
    fetchBenchmark()
      .then(setReport)
      .catch((err) => console.error("benchmark fetch failed", err));
  }, []);

  if (!report) {
    return (
      <Panel title="// benchmark · terminal_collapse" meta="[ REPLAY ]">
        <div className="mono" style={{ fontSize: 12, color: "var(--text-muted)", padding: "20px 0" }}>
          loading scenario…
        </div>
      </Panel>
    );
  }

  const terminal = report.scenarios.find((s) => s.name === "terminal-collapse");
  if (!terminal) return null;
  const ai = terminal.results.find((r) => r.strategyName === "solvent-ai");
  if (!ai) return null;

  return (
    <Panel title="// benchmark · terminal_collapse" meta={`[ ${ai.log.length} TICKS ]`}>
      <div className="mono" style={{ fontSize: 11.5 }}>
        {ai.log.map((entry: DecisionLogEntry) => {
          const valueUsd = Number(entry.valueAfter) / 1_000_000; // safe-asset has 6 decimals (USDC)
          const colour = REASON_COLOUR[entry.reasonCode] ?? "var(--ink-cyan-bright)";
          return (
            <div
              key={entry.timestamp}
              style={{
                display: "flex",
                gap: 14,
                padding: "6px 0",
                borderBottom: "1px solid rgba(124,213,255,.08)",
                alignItems: "center",
              }}
            >
              <span style={{ opacity: 0.45, minWidth: 56 }}>t+{(entry.timestamp / 3600).toFixed(0)}h</span>
              <span style={{ color: colour, minWidth: 130 }}>{entry.reasonCode}</span>
              <span style={{ flex: 1, color: "var(--text-muted)" }}>regime={entry.regime}  ·  action={entry.action}</span>
              <span style={{ color: "var(--text-strong)" }}>${valueUsd.toLocaleString("en-US", { maximumFractionDigits: 0 })}</span>
            </div>
          );
        })}
      </div>
    </Panel>
  );
}
